    0x0b, 0x07, // Scan-Limit Register, 0x07 to show all lines

};

void resetMatrix()
{
  for(int i = 1;i<=8;i++)
    {
      digitalWrite(CS,LOW);
        shiftOut(DIN,CLK,MSBFIRST,i);
        shiftOut(DIN,CLK,MSBFIRST,0x00);
      digitalWrite(CS,HIGH);
    }
}
